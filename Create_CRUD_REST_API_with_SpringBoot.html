Building a CRUD RESTfull API with SpringBoot

Date mar. 30 janvier 2018 Modified mar. 30 janvier 2018 Tags Spring / SpringBoot / REST / API

This post will walk through creating a CRUD REST API with SpringBoot framework. For this demo, I use an in-memory database (H2) and Spring Data REST which is a hypermedia-based RESTful front end.

  • Create the directory structure
$ mkdir -p ContactDirectory/src/main/java/ContactDirectory
$ cd ContactDirectory
  • Create a Gradle build file
$ nano build.gradle

buildscript {
    repositories {
        mavenCentral()
    }
    dependencies {
        classpath("org.springframework.boot:spring-boot-gradle-plugin:1.5.9.RELEASE")
    }
}

apply plugin: 'java'
apply plugin: 'eclipse'
apply plugin: 'idea'
apply plugin: 'org.springframework.boot'

jar {
    baseName = 'ContactDirectory-rest-service'
    version =  '0.1.0'
}

repositories {
    mavenCentral()
}

sourceCompatibility = 1.8
targetCompatibility = 1.8

dependencies {
    compile("org.springframework.boot:spring-boot-starter-web")
    compile("org.springframework.boot:spring-boot-starter-data-rest")
    compile("org.springframework.boot:spring-boot-starter-data-jpa")
    compile("com.h2database:h2")
}
  • Create an Object
$ nano src/main/java/ContactDirectory/Person.java

package ContactDirectory;

import javax.persistence.Entity;
import javax.persistence.GeneratedValue;
import javax.persistence.GenerationType;
import javax.persistence.Id;

@Entity
public class Person {

    @Id
    @GeneratedValue(strategy = GenerationType.AUTO)
    private long id;

    private String firstName;
    private String lastName;

    public String getFirstName() {
        return firstName;
    }

    public void setFirstName(String firstName) {
        this.firstName = firstName;
    }

    public String getLastName() {
        return lastName;
    }

    public void setLastName(String lastName) {
        this.lastName = lastName;
    }
}
  • Create a repository
$ nano src/main/java/ContactDirectory/PersonRepository.java

package ContactDirectory;

import java.util.List;

import org.springframework.data.repository.CrudRepository;
import org.springframework.data.repository.query.Param;
import org.springframework.data.rest.core.annotation.RepositoryRestResource;

@RepositoryRestResource
public interface PersonRepository extends CrudRepository<Person, Long> {

}
  • Creates a standalone application
$ nano src/main/java/ContactDirectory/Application.java

package ContactDirectory;

import org.springframework.boot.SpringApplication;
import org.springframework.boot.autoconfigure.SpringBootApplication;

@SpringBootApplication
public class Application {

    public static void main(String[] args) {
        SpringApplication.run(Application.class, args);
    }
}
  • Install gradle wraper in your project
$ gradle-3.4.1 wrapper
  • Build
$ ./gradlew build
  • Run
$ java -jar build/libs/ContactDirectory-rest-service-0.1.0.jar
  • Test
$ curl http://localhost:8080
$ curl http://localhost:8080/persons
$ curl -i -X POST -H "Content-type:application/json" -d '{ "firstName" : "Didier", "lastName" : "BERNAUDEAU" }' http://localhost:8080/persons
$ curl http://localhost:8080/persons/1
